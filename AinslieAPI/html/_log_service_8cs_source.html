84 public LogEntry InsertLog(LogTypes logType, int threadID, LogSources logSource, params object[] DataList)
85 {
86 LogEntry logEntry;
87 try
88 {
89 //try to create log entry by deserialising objects in param array datalist
90 logEntry = new LogEntry()
91 {
92 Created = DateTime.Now,
93 Type = logType.ToString(),
94 ThreadID = threadID,
95 Source = logSource.ToString(),
96 Data = JsonConvert.SerializeObject(DataList, new JsonSerializerSettings
97 {
98 ReferenceLoopHandling = ReferenceLoopHandling.Ignore
99 })
100 };
101 }
102 catch (Exception ex)
103 {
104 //create new logentry with exception
105 logEntry = new LogEntry()
106 {
107 Created = DateTime.Now,
108 Type = LogTypes.Exception.ToString(),
109 ThreadID = threadID,
110 Source = LogSources.LogService.ToString(),
111 Data = $"Exception creating logEntry: {ex.ToString()}"
112 };
113 }
114 try
115 {
116 _context.LogEntries.Add(logEntry);
117 _context.SaveChanges();
118 return logEntry;
119 }
120 catch (Exception ex)
121 {
122 WriteLog(LogTypes.Exception, "LogService", $"Exception trying to insert log. Exception: {ex.ToString()}");
123 }
124 return null;
125 }

130 private void WriteLog(LogTypes Type, string Source, string Description)
131 {
132 var LogPath = $"{AppDomain.CurrentDomain.BaseDirectory}/Log/";
133 if (!Directory.Exists(LogPath))
134 Directory.CreateDirectory(LogPath);
135 var LogFile = $"AinslieAPI.Log";
136 try
137 {
138 StreamWriter SW = File.AppendText(LogPath + LogFile);
139 SW.WriteLine($"{DateTime.Now.ToString()} [{Type.ToString()}] {Description}");
140 SW.Close();
141 }

143 {
144 }

147 public object TrimLogs()
148 {
149 try
150 {
151 _context.Database.SetCommandTimeout(TimeSpan.FromMinutes(2));
152 var deleteCount = _context.Database.ExecuteSqlRaw("DELETE FROM LogEntries WHERE Created < @cutOff", new SqlParameter("@cutOff", DateTime.Now.AddDays(-_settings.GeneralSettings.DaysToKeepLogs)));
153 return new { error = false, message = $"deleted {deleteCount} logs" };
154 }
155 catch (Exception ex)
