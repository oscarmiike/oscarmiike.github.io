67 [HttpPost("getlogs")]
68 public IActionResult GetLogs([FromBody] LogRequest logRequest)
69 {
70 DateTime fromDateTimeUTC = DateTime.MinValue;
71 if (logRequest.fromUnixUTCMilli > 0)
72 {
73 fromDateTimeUTC = DateTimeOffset.FromUnixTimeMilliseconds(logRequest.fromUnixUTCMilli).DateTime;
74 }
75 DateTime fromDateTimeLocal = fromDateTimeUTC.ToLocalTime();
76 var dateRangeStart = logRequest.dateRangeStartUTC.ToLocalTime();
77 if (logRequest.startHour > -1)
78 dateRangeStart = dateRangeStart.AddHours(logRequest.startHour);
79 var dateRangeEnd = logRequest.dateRangeEndUTC.ToLocalTime();
80
81 var logsQuery = _context.LogEntries.AsNoTracking()
82 .Where(le => le.Created >= dateRangeStart && le.Created <= dateRangeEnd)
83 .AsEnumerable();
84
85 if (logRequest.initial)
86 {
87 logsQuery = logsQuery
88 .Where(le => logRequest.logType == null || logRequest.logType.All(string.IsNullOrEmpty) || logRequest.logType.Any(lt => le.Type.ToLower() == lt.ToLower()))
89 .Where(le => logRequest.logSource == null || logRequest.logSource.All(string.IsNullOrEmpty) || logRequest.logSource.Any(ls => le.Source.ToLower() == ls.ToLower()))
90 .Where(le => string.IsNullOrEmpty(logRequest.searchString) || le.Data.ToLower().Contains(logRequest.searchString.ToLower()));
91
92 if (logRequest.startHour > -1)
93 //if specifying a start hour, take logs from the start hour, then order by descending
94 logsQuery = logsQuery.Take(logRequest.maxLogs).OrderByDescending(le => le.Created).ToList();
95 else
96 //if not specifying start hour, order by descending and then take latest
97 logsQuery = logsQuery.OrderByDescending(le => le.Created).Take(logRequest.maxLogs).ToList();
98 }
99 else
100 {
101 logsQuery = logsQuery
102 .Where(le => le.Created > fromDateTimeLocal)
103 .Where(le => logRequest.logType == null || logRequest.logType.All(string.IsNullOrEmpty) || logRequest.logType.Any(lt => le.Type.ToLower() == lt.ToLower()))
104 .Where(le => logRequest.logSource == null || logRequest.logSource.All(string.IsNullOrEmpty) || logRequest.logSource.Any(ls => le.Source.ToLower() == ls.ToLower()))
105 .Where(le => string.IsNullOrEmpty(logRequest.searchString) || le.Data.ToLower().Contains(logRequest.searchString.ToLower()))
106 .OrderByDescending(le => le.Created);
107 }
108 if (!logRequest.initial && logsQuery.Count() < 1)
109 {
110 return Ok();
111 }
112 return Ok(logsQuery);
113 }
