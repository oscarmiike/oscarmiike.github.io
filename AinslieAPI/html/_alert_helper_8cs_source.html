67 private List<LogEntry>? FetchAndProcessLogs(string apiSource, ApiSource config)
68 {
69 var maxWindowMinutes = (_logNotificationsSettings.LogNotifications?.ApiSources?
70 .SelectMany(apiSrc => (apiSrc.LogSourceNotifications?.Select(ln => ln.WindowMinutes) ?? new List<int>())
71 .Append(apiSrc.DefaultNotification.WindowMinutes))
72 .Max()) ?? 0;
73
74 var allLogTypes = (config.DefaultNotification?.LogTypes?.Select(lt => lt.LogType) ?? new List<string>())
75 .Union(config.LogSourceNotifications?.SelectMany(ln => ln.LogTypes?.Select(lt => lt.LogType) ?? new List<string>()) ?? new List<string>())?
76 .ToList() ?? new List<string>();
77
78 if (maxWindowMinutes == 0 || !allLogTypes.Any())
79 return null;
80
81 var UTCTimeNow = DateTime.UtcNow;
82 var logRequest = new LogRequest
83 {
84 dateRangeStartUTC = UTCTimeNow.AddMinutes(-maxWindowMinutes),
85 dateRangeEndUTC = UTCTimeNow,
86 logType = allLogTypes,
87 maxLogs = 100,
88 startHour = -1,
89 logSource = new List<string>(),
90 searchString = ""
91 };
92
93 // Check if the apiSource is TaskAPI
94 if (apiSource.Equals("TaskAPI", StringComparison.OrdinalIgnoreCase))
95 {
96 var logs = _logService.FetchLogs(logRequest);
97 return logs;
98 }
99 else
100 {
101 // Use InternalAPIService for other ApiSources
102 var logs = _iapiService.GetUnifiedLogs(logRequest, apiSource);

106 // return new List<LogEntry>();

117 private List<Alert> CheckLogsAgainstThresholds(List<LogEntry> logs, ApiSource config)
118 {
119 var alerts = new List<Alert>();
120 //var dateTime = DateTime.UtcNow.ToLocalTime();
121
122 if (logs == null || !logs.Any())
123 {
124 _logService.LogInformation(LogService.LogSources.AlertHelper, "No logs were fetched.");
125 return alerts;
126 }
127
128 // Apply Specific Log Source Notification settings
129
130 foreach (var logSourceNotification in config.LogSourceNotifications ?? new List<LogSourceNotification>())
131 {
132 // get logs that are in the specified log source, and are of a type that exists in the specified notification config, and fall withing the specified time window
133 var specificNotificationLogs = logs.Where(log =>
134 log.Source == logSourceNotification.Name &&
135 (logSourceNotification.LogTypes?.Any(lt => lt.LogType == log.Type) ?? false) &&
136 log.Created >= DateTime.Now.AddMinutes(-logSourceNotification.WindowMinutes));
137 if (specificNotificationLogs != null && specificNotificationLogs.Any())
138 {
139 // get logs for each type specified in notification config
140 foreach (var logType in logSourceNotification.LogTypes ?? new List<LogTypeEntry>())
141 {
142 var relevantLogs = specificNotificationLogs.Where(log => log.Type == logType.LogType);
143 // if enough logs to satisfy notification type count, create an alert
144 if (relevantLogs != null && relevantLogs.Count() >= logType.Count)
145 {
146 alerts.Add(new Alert
147 {
148 LogSource = logSourceNotification.Name,
149 LogType = logType.LogType,
150 Count = relevantLogs.Count(),
151 TriggeringLogs = relevantLogs.ToList()
152 });
153 }
154 }
155 }
156 }
157
158 // Apply DefaultNotification settings
159
160 // get logs that are not in any specified log sources, and are of a type that exists in the default notification config, and fall withing the default time window
161 var defaultNotificationLogs = logs.Where(log =>
162 (!config.LogSourceNotifications?.Any(lsn => lsn.Name == log.Source) ?? true) &&
163 config.DefaultNotification.LogTypes.Any(lt => lt.LogType == log.Type) &&
164 log.Created >= DateTime.Now.AddMinutes(-config.DefaultNotification.WindowMinutes));
165 if (defaultNotificationLogs != null && defaultNotificationLogs.Any())
166 {
167 // get logs for each type specified in default notification settings
168 foreach (var logType in config.DefaultNotification?.LogTypes ?? new List<LogTypeEntry>())
169 {
170 var relevantLogs = defaultNotificationLogs.Where(log => log.Type == logType.LogType);
171 // if enough logs to satisfy default notification type count, create an alert
172 if (relevantLogs != null && relevantLogs.Count() >= logType.Count)
173 {
174 alerts.Add(new Alert
175 {
176 LogSource = "Default",
177 LogType = logType.LogType,
178 Count = relevantLogs.Count(),
179 TriggeringLogs = relevantLogs.ToList()
180 });
181 }
182 }
183 }
184 return alerts;

193 private List<string> SendAlerts(List<Alert> alerts, string apiSource, ApiSource config)
194 {
195 var sentAlertDetails = new List<string>();
196 var failedAlerts = new List<string>();
197 var dateTime = DateTime.UtcNow.ToLocalTime();
198
199 foreach (var alert in alerts)
200 {
201 try
202 {
203 // Determine the notification minimum frequency for the alert
204 int notificationMinimumFrequency = config.DefaultNotification.NotificationMinimumFrequency;
205 var specificLogSourceNotification = config.LogSourceNotifications?
206 .FirstOrDefault(lsn => lsn.Name == alert.LogSource);
207
208 if (specificLogSourceNotification != null)
209 {
210 notificationMinimumFrequency = specificLogSourceNotification.NotificationMinimumFrequency;
211 }
212 if (notificationMinimumFrequency == 0)
213 continue;
214
215 // Check if notification frequency conditions are met - if null lastNotification, send alert
216 var lastNotification = _context.AlertNotifications.AsNoTracking()
217 .Where(n => n.ApiSource == apiSource && n.LogSource == alert.LogSource && n.LogType == alert.LogType)
218 .OrderByDescending(n => n.LastNotification)
219 .FirstOrDefault();
220
221 if (lastNotification != null && (dateTime - lastNotification.LastNotification).TotalMinutes < notificationMinimumFrequency)
222 {
223 // Skip sending the alert as the minimum frequency condition is not met
224 continue;
225 }
226
227 var subject = $"{apiSource} Alert: {alert.Count} {alert.LogType} logs detected from {alert.LogSource}";
228 var logsDetailsJson = JsonConvert.SerializeObject(alert.TriggeringLogs, Formatting.Indented);
229 var logsDetailsHtml = FormatJsonAsHtml(logsDetailsJson); // Convert JSON to HTML
230 var body = $"Triggering Logs:<br>{logsDetailsHtml}"; // Use the HTML in the email body
231
232 foreach (string emailAddress in _logNotificationsSettings.LogNotifications?.GeneralSettings?.NotificationEmails ?? new List<string>())
233 {
234 _iapiService.SendEmail(new EmailPost()
235 {
236 NameFrom = _settings.GeneralSettings.NotificationEmailName,
237 EmailFrom = _settings.GeneralSettings.NotificationEmailFrom,
238 EmailTo = emailAddress,
239 Subject = subject,
240 HTMLBody = body
241 });
242 }
243
244 // add new each time to keep a basic history
245 _context.AlertNotifications.Add(new AlertNotification
246 {
247 ApiSource = apiSource,
248 LogSource = alert.LogSource,
249 LogType = alert.LogType,
250 LastNotification = dateTime
251 });
252
253 _context.SaveChanges();
254
255 sentAlertDetails.Add($"Sent alert: {subject}");
256 _logService.LogInformation(LogService.LogSources.AlertHelper, $"Sent alert: {subject}");
257 }
258 catch (Exception ex)
259 {
260 var errorSubject = $"Failed to send {apiSource} alert: {alert.Count} {alert.LogType} logs in {alert.LogSource}";
261 _logService.LogError(LogService.LogSources.AlertHelper, $"{errorSubject}. Error: {ex}");
262 failedAlerts.Add(errorSubject);
263 }
264 }
265
266 if (failedAlerts.Any())
267 {
268 sentAlertDetails.Add($"Failed to send some alerts. See logs for details.");
269 }
270
271 return sentAlertDetails;
272 }

277 private string FormatJsonAsHtml(string json)
278 {
279 var sb = new StringBuilder();
280 sb.AppendLine("<table style='border-collapse: collapse; width: 100%; font-family: Arial, sans-serif; font-size: 14px;'>");
281
282 var jsonArray = JArray.Parse(json);
283 bool isGrey = true;
284
285 foreach (var jt in jsonArray.Children<JObject>())
286 {
287 var bgColor = isGrey ? "#f2f2f2" : "#ffffff";
288 sb.AppendLine($"<tr style='background-color: {bgColor};'>");
289 sb.Append("<td style='border: 1px solid #dddddd; text-align: left; padding: 8px;'><pre>{");
290
291 foreach (var jprop in jt.Properties())
292 {
293 string formattedValue = FormatJsonValue(jprop.Value);
294 sb.Append($"<br> \"{System.Net.WebUtility.HtmlEncode(jprop.Name)}\": {formattedValue}");
295 if (jprop != jt.Properties().Last())
296 {
297 sb.Append(",");
298 }
299 }
300
301 sb.Append("<br>}</pre></td>");
302 sb.AppendLine("</tr>");
303
304 isGrey = !isGrey;
305 }
306
307 sb.AppendLine("</table>");
308 return sb.ToString();
309 }

311 private string FormatJsonValue(JToken value)
312 {
313 switch (value.Type)
314 {
315 case JTokenType.String:
316 return $"\"<span style='color:green;'>{System.Net.WebUtility.HtmlEncode(value.ToString())}</span>\"";
317 case JTokenType.Integer:
318 case JTokenType.Float:
319 return $"<span style='color:purple;'>{value}</span>";
320 case JTokenType.Date:
321 return $"\"<span style='color:blue;'>{value}</span>\"";
322 default:
323 return System.Net.WebUtility.HtmlEncode(value.ToString());
324 }
325 }
