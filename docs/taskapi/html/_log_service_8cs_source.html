92 public LogEntry InsertLog(LogTypes logType, int threadID, LogSources logSource, params object[] DataList)
93 {
94 LogEntry logEntry;
95 try
96 {
97 //try to create log entry by deserialising objects in param array datalist
98 logEntry = new LogEntry()
99 {
100 Created = DateTime.Now,
101 Type = logType.ToString(),
102 ThreadID = threadID,
103 Source = logSource.ToString(),
104 Data = JsonConvert.SerializeObject(DataList, new JsonSerializerSettings
105 {
106 ReferenceLoopHandling = ReferenceLoopHandling.Ignore
107 })
108 };
109 }
110 catch (Exception ex)
111 {
112 //create new logentry with exception
113 logEntry = new LogEntry()
114 {
115 Created = DateTime.Now,
116 Type = LogTypes.Exception.ToString(),
117 ThreadID = threadID,
118 Source = LogSources.LogService.ToString(),
119 Data = $"Exception creating logEntry: {ex.ToString()}"
120 };
121 }
122 try
123 {
124 _context.LogEntries.Add(logEntry);
125 _context.SaveChanges();
126 return logEntry;
127 }
128 catch (Exception ex)
129 {
130 WriteLog(LogTypes.Exception, "LogService", $"Exception trying to insert log. Exception: {ex.ToString()}");
131 }
132 return null;
133 }

170 public List<LogEntry> FetchLogs(LogRequest logRequest)
171 {
172 DateTime fromDateTimeUTC = DateTime.MinValue;
173 if (logRequest.fromUnixUTCMilli > 0)
174 {
175 fromDateTimeUTC = DateTimeOffset.FromUnixTimeMilliseconds(logRequest.fromUnixUTCMilli).DateTime;
176 }
177 DateTime fromDateTimeLocal = fromDateTimeUTC.ToLocalTime();
178 var dateRangeStart = logRequest.dateRangeStartUTC.ToLocalTime();
179 if (logRequest.startHour > -1)
180 {
181 dateRangeStart = dateRangeStart.AddHours(logRequest.startHour);
182 }
183 var dateRangeEnd = logRequest.dateRangeEndUTC.ToLocalTime();
184
185 var logsQuery = _context.LogEntries.AsNoTracking()
186 .Where(le => le.Created >= dateRangeStart && le.Created <= dateRangeEnd)
187 .AsEnumerable();
188
189 if (logRequest.initial)
190 {
191 logsQuery = logsQuery
192 .Where(le => logRequest.logType == null || logRequest.logType.All(string.IsNullOrEmpty) || logRequest.logType.Any(lt => le.Type.ToLower() == lt.ToLower()))
193 .Where(le => logRequest.logSource == null || logRequest.logSource.All(string.IsNullOrEmpty) || logRequest.logSource.Any(ls => le.Source.ToLower() == ls.ToLower()))
194 .Where(le => string.IsNullOrEmpty(logRequest.searchString) || le.Data.ToLower().Contains(logRequest.searchString.ToLower()));
195
196 if (logRequest.startHour > -1)
197 {
198 // If specifying a start hour, take logs from the start hour, then order by descending
199 return logsQuery.Take(logRequest.maxLogs).OrderByDescending(le => le.Created).ToList();
200 }
201 else
202 {
203 // If not specifying start hour, order by descending and then take latest
204 return logsQuery.OrderByDescending(le => le.Created).Take(logRequest.maxLogs).ToList();
205 }
206 }
207 else
208 {
209 logsQuery = logsQuery
210 .Where(le => le.Created > fromDateTimeLocal)
211 .Where(le => logRequest.logType == null || logRequest.logType.All(string.IsNullOrEmpty) || logRequest.logType.Any(lt => le.Type.ToLower() == lt.ToLower()))
212 .Where(le => logRequest.logSource == null || logRequest.logSource.All(string.IsNullOrEmpty) || logRequest.logSource.Any(ls => le.Source.ToLower() == ls.ToLower()))
213 .Where(le => string.IsNullOrEmpty(logRequest.searchString) || le.Data.ToLower().Contains(logRequest.searchString.ToLower()))
214 .OrderByDescending(le => le.Created);
215
216 return logsQuery.ToList();
217 }
218 }
