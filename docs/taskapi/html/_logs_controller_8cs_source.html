115 _logService.LogInformation(LogService.LogSources.LogsController, "task api log trim successful", result);

132 [HttpGet("CheckForAlerts")]
133 public IActionResult CheckForAllAlerts()
134 {
135 var processedSources = new List<string>();
136 var failedSources = new List<string>();
137
138 if (_logNotificationsSettings?.LogNotifications?.ApiSources == null)
139 {
140 _logService.LogError(LogService.LogSources.LogsController, $"Error LogNotificationsSettings.{_env.EnvironmentName}.json not found");
141 return StatusCode(500, new { Message = "LogNotificationsSettings not found." });
142 }
143
144 foreach (var apiConfig in _logNotificationsSettings.LogNotifications.ApiSources)
145 {
146 try
147 {
148 // Use AlertHelper to process the API source
149 _alertHelper.ProcessApiSource(apiConfig);
150 processedSources.Add(apiConfig.Name);
151 }
152 catch (Exception ex)
153 {
154 _logService.LogException(LogService.LogSources.LogsController, $"Error processing {apiConfig.Name}: {ex}");
155 failedSources.Add(apiConfig.Name);
156 }
157 }
158
159 if (failedSources.Any())
160 {
161 return StatusCode(500, new { Message = "Some API sources failed to process.", ProcessedSources = processedSources, FailedSources = failedSources });
162 }
163
164 return Ok(new { Message = $"All API sources processed for alerts. -- {_env.EnvironmentName}", ProcessedSources = processedSources });
165 }
