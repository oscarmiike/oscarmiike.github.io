237 {
238 List<BaseSpot> spotList = new List<BaseSpot>();
239 foreach (var x in _settings.PolygonOracleSettings.Assets)
240 {
241 double spot = 0;
242 string pairCode = "USD";
243 //if market open, pull data and create basespot with live data, otherwise create dummy basespot with 0
244 if (CanSync())
245 {
246 LatestRoundDataResponse latestRoundResponse = _alchemyService.LatestRoundData(x.Contract);
247 _logService.LogDebug(LogService.LogSources.SpotHelper, $"latest round data from polygon oracle for {x.AssetCode}", latestRoundResponse);
248 if (latestRoundResponse == null)
249 {
250 _logService.LogError(LogService.LogSources.SpotHelper, "response from polygon oracle service is null");
251 }
252 else
253 {
254 try
255 {
256 //check feed time variation
257 var feedTime = DateTimeOffset.FromUnixTimeSeconds(long.Parse(latestRoundResponse.updatedAt.ToString())).UtcDateTime.ToLocalTime();
258 if ((pullTime - feedTime).TotalMinutes > _settings.SpotFeedSettings.TimeDifferenceThresholdMinutes)
259 {
260 _logService.LogError(LogService.LogSources.SpotHelper, "polygon oracle time variation is above threshold");
261 }
262 else
263 {
264 spot = Convert.ToDouble(latestRoundResponse.answer.ToString()) / Math.Pow(10, _settings.PolygonOracleSettings.ContractDecimals);

266 }

269 _logService.LogException(LogService.LogSources.SpotHelper, "exception reading values from polygon oracle response", ex);
270 }
271 }
272 }

274 {
275 AssetCode = x.AssetCode,
276 PairCode = pairCode,
277 Per = x.AssetCode == "AUD" ? "$" : "oz",
278 Source = _settings.PolygonOracleSettings.SourceName,
279 PullTime = pullTime,
280 Spot = spot
281 };
282 spotList.Add(bs);
283 }
284 return spotList;

385 }
386 try
387 {
388 try
389 {
390 //check feed time variation

400 }
401 //try first ask
402 if (!double.TryParse(oandaAssetResponse?.prices?.FirstOrDefault()?.asks?.OrderBy(ask => ask.price).FirstOrDefault()?.price ?? "0", out spot) || spot == 0)
403 {
404 //try closeoutask

406 {
407 //try first bid
408 if (!double.TryParse(oandaAssetResponse?.prices?.FirstOrDefault()?.bids?.OrderByDescending(bid => bid.price).FirstOrDefault()?.price ?? "0", out spot) || spot == 0)
409 {
410 _logService.LogError(LogService.LogSources.SpotHelper, $"unable to find suitable price in oanda feed for {x.AssetCode}", oandaAssetResponse);
411 }
412 }
413 }
414 }
415 catch (Exception ex)
416 {
417 _logService.LogException(LogService.LogSources.SpotHelper, $"exception getting spot from oanda feed for {x.AssetCode}", oandaAssetResponse, ex);
418 }
419 BaseSpot bs = new BaseSpot()
420 {
421 AssetCode = x.AssetCode,
422 PairCode = "USD",
423 Per = x.AssetCode == "AUD" ? "$" : "oz",
424 Source = _settings.OandaSettings.SourceName,

595 public (bool Error, object? Result) SaveDailySpot()
596 {
597 bool isDebug = _env.IsDevelopment() || _env.IsStaging();
598 try
599 {
600 int Year = DateTime.Now.Year;
601 int Month = DateTime.Now.Month;
602 int Day = DateTime.Now.Day;
603 //5 mins before is ok, allow 1 hr after in case retries are required
604 var WindowOpen = new DateTime(Year, Month, Day, 15, 55, 0);
605 var WindowClose = new DateTime(Year, Month, Day, 16, 05, 0);
606 if (!isDebug && !(DateTime.Now > WindowOpen & DateTime.Now < WindowClose))
607 return (false, new { success = false, message = "request does not fall within daily save time window" });
608 //check if done today already
609 if (!isDebug && _context.DailyCloses.Where(dc => dc.Year == Year && dc.Month == Month && dc.Day == Day).Count() > 0)
610 return (false, new { success = false, message = "daily closes already completed for today", Data = _context.DailyCloses.Where(dc => dc.Year == Year && dc.Month == Month && dc.Day == Day).ToList() });
611 //get current asset list from priceapp
612 var assetList = _iapiService.PriceappAssets();
613 if (assetList == null || assetList.Count == 0)
614 {
615 return (true, new { success = false, message = "asset list from priceapp was null" });
616 }
617 List<DailyClose> dailyCloses = new List<DailyClose>();
618 //create a daily close for AUD
619 var first = assetList.FirstOrDefault();
620 var audusd = Math.Truncate(10000 * first.audusd) / 10000;
621 DailyClose AUDClose = new DailyClose()
622 {
623 Year = Year,
624 Month = Month,
625 Day = Day,
626 AssetCode = "AUD",
627 PairCode = "USD",
628 Per = "$",
629 Buy = audusd,
630 Spot = audusd,
631 Sell = audusd
632 };
633 dailyCloses.Add(AUDClose);
634 //create daily close for all else
635 foreach (PriceappAsset x in assetList)
636 {
637 if (x.assetCode == "XX1")
638 continue;
639 // all assets from internal api are AUD
640 DailyClose dc = new DailyClose()
641 {
642 Year = Year,
643 Month = Month,
644 Day = Day,
645 AssetCode = x.assetCode,
646 PairCode = "AUD",
647 Per = x.spotUnit?.unitName ?? "" ,
648 Buy = x.environmentBuy,
649 Spot = x.environmentSpot,
650 Sell = x.environmentSell
651 };
652 dailyCloses.Add(dc);
653 }
654 _context.DailyCloses.AddRange(dailyCloses);

669 private (bool Error, object? Result) ExportChartData()
670 {
671 //used for name mapping but also used for ordering results
672 Dictionary<string, string> assetCodeToNameMap = new Dictionary<string, string>
673 {
674 ["XAU"] = "gold",
675 ["XAG"] = "silver",
676 ["XPT"] = "platinum",
677 ["AUD"] = "AUD"
678 };
679 //use iapiservice to get chart data, order it by timestamp

681 if (dailyCloseChartData == null || dailyCloseChartData.Count() < 1)
682 {
683 _logService.LogError(LogService.LogSources.SpotHelper, "could not export chart data, response from internal api service was null or empty");

685 }
686 //group by metal, order by assetcodetonamemap dictionary elements, flatten the igrouping
687 var dailyCloseChartDataSorted = dailyCloseChartData
688 .GroupBy(ci => new { metal = ci.Metal, order = assetCodeToNameMap.ToList().IndexOf(assetCodeToNameMap.First(map => map.Key == ci.Metal)) })
689 .OrderBy(grp => grp.Key.order)
690 .SelectMany(grp => grp);
691 //select new list for AUD currency convert and asset code to name map at same time
692 var dailyCloseChartDataAUD = dailyCloseChartDataSorted.ToList().Select(ci => new DailyCloseChartItem()
693 {
694 Currency = "AUD",
695 Metal = assetCodeToNameMap[ci.Metal],
696 Spot = decimal.Round(ConvertChartDataCurrency("AUD", ci, dailyCloseChartData), 4),
697 Timestamp = ci.Timestamp
698 });
699 //select new list for USD currency convert and asset code to name map at same time
700 var dailyCloseChartDataUSD = dailyCloseChartDataSorted.ToList().Select(ci => new DailyCloseChartItem()
701 {
702 Currency = "USD",
703 Metal = assetCodeToNameMap[ci.Metal],
704 Spot = decimal.Round(ConvertChartDataCurrency("USD", ci, dailyCloseChartData), 4),
705 Timestamp = ci.Timestamp
706 });
707 //save files
708 try
709 {
710 if (!Directory.Exists(_settings.PriceappSettings.ChartDataExportPath))
711 Directory.CreateDirectory(_settings.PriceappSettings.ChartDataExportPath);
712 File.WriteAllText($"{_settings.PriceappSettings.ChartDataExportPath}ChartData_AUD.json", JsonConvert.SerializeObject(dailyCloseChartDataAUD, Formatting.None));
713 File.WriteAllText($"{_settings.PriceappSettings.ChartDataExportPath}ChartData_USD.json", JsonConvert.SerializeObject(dailyCloseChartDataUSD, Formatting.None));
714 _logService.LogDebug(LogService.LogSources.SpotHelper, "chart data exported");
715 return (false, $"chart data exported");
716 }

727 private decimal ConvertChartDataCurrency(string convertTo, DailyCloseChartItem convertItem, List<DailyCloseChartItem> metalPrices)
728 {
729 if (convertTo.ToLower() == convertItem.Currency.ToLower() || convertItem.Metal.ToLower() == "aud")
730 return convertItem.Spot;
731 var audUsd = metalPrices.FirstOrDefault(mp => mp.Timestamp == convertItem.Timestamp && mp.Metal.ToLower() == "aud")?.Spot;
732 if (audUsd == null)
733 {
734 //find closest with later date as preference
735 audUsd = metalPrices.Where(mp => mp.Metal.ToLower() == "aud" && mp.Timestamp > convertItem.Timestamp)?.OrderBy(mp => mp.Timestamp).FirstOrDefault()?.Spot;
736 if (audUsd == null)
737 {
738 //find closest with earlier date
739 audUsd = metalPrices.Where(mp => mp.Metal.ToLower() == "aud" && mp.Timestamp < convertItem.Timestamp)?.OrderByDescending(mp => mp.Timestamp).FirstOrDefault()?.Spot;
740 }
741 }
742 if (audUsd != null)
743 {
744 switch (convertTo.ToLower())
745 {
746 case "aud":
747 return convertItem.Spot / (decimal)audUsd;
748 default:
749 return convertItem.Spot * (decimal)audUsd;
750 }
751 }
752 return convertItem.Spot;
753 }
