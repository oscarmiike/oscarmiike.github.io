87 private List<AssetPrice> RefreshPriceappAssets()
88 {
89 //try pull from cacheddata
90 var assetListCache = _cachedData.CacheItems.Where(ci => ci.CacheType == CacheType.AssetList).FirstOrDefault();
91 //if nothing in cacheddata create new one with correct type and add to cacheddata
92 if (assetListCache == null)
93 {
94 assetListCache = new CacheItem();
95 assetListCache.CacheType = CacheType.AssetList;
96 _cachedData.CacheItems.Add (assetListCache);
97 }
98 //if refresh conditions met - either data is null or data expired
99 if (assetListCache.Data == null || assetListCache.Expires < DateTime.Now)
100 {
101 _logService.LogInformation(LogService.LogSources.CacheDataHelper, $"refreshing asset price list cache");
102 bool lockTaken = false;
103 Monitor.TryEnter(Padlock_AssetPrice, ref lockTaken);
104 if (lockTaken)
105 {
106 try
107 {
108 if (!assetListCache.Refreshing)
109 {
110 assetListCache.Refreshing = true;
111 assetListCache.Data = _iapiService.GetAssetPricing() ?? new List<AssetPrice>();
112 assetListCache.Expires = DateTime.Now.AddSeconds(_appSettings.CacheDataSettings.AssetListCacheSec);
113 }
114 }
115 finally
116 {
117 assetListCache.Refreshing = false;
118 Monitor.Exit(Padlock_AssetPrice);
119 }
120 }
121 }
122 try
123 {
124 return (List<AssetPrice>?)assetListCache.Data ?? new List<AssetPrice>();
125 }
126 catch (Exception)
127 {
128 return new List<AssetPrice>();
