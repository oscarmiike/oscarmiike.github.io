50 [HttpGet("{branch}/newverification")]
51 [HttpPost("{branch}/newverification")]
52 [AllowAnonymous]
53 public IActionResult NewVerification(string branch, [FromQuery] string userid = "")
54 {
55 _logger.LogInformation(LogService.LogSources.GreenIdController, $"request received on greenid newverification endpoint branch: {branch} userid: {userid}");
56 // get customer object from green id and add to local greeniddb via internalapi
57 GreenIdCustomer greenIDCustomer = _iaService.GetGreenIdCustomer(branch, userid);
58 // post customer to internal api netsuite endpoint for add or update
59 if (greenIDCustomer != null)
60 {
61 if (greenIDCustomer.error || string.IsNullOrEmpty(greenIDCustomer.verificationId))
62 {
63 _logger.LogError(LogService.LogSources.GreenIdController, $"error getting customer from green id", greenIDCustomer);
64 }
65 _logger.LogDebug(LogService.LogSources.GreenIdController, $"customer retrieved from green id", greenIDCustomer);
66 NetsuiteCustomerRequest netsuiteCustomerRequest = _netsuiteHelper.CreateNetsuiteCustomerRequest(branch, greenIDCustomer);
67 if (netsuiteCustomerRequest != null)
68 {
69 var netsuiteInsertId = _iaService.PostNetsuiteCustomer(netsuiteCustomerRequest);
70 if (netsuiteInsertId == 0)
71 {
72 _logger.LogError(LogService.LogSources.GreenIdController, $"error inserting customer to netsuite", greenIDCustomer);
73 }
74 else
75 {
76 _logger.LogDebug(LogService.LogSources.GreenIdController, $"customer posted to netsuite successfully. id: {netsuiteInsertId}");
77 var updateNetsuiteIdOk = _iaService.UpdateGreenIdDbNetsuiteId(greenIDCustomer.verificationId, netsuiteInsertId);
78 if(!updateNetsuiteIdOk)
79 _logger.LogError(LogService.LogSources.GreenIdController, $"error updating greeniddb netsuite customer id: {greenIDCustomer.verificationId}/{netsuiteInsertId}");
80 }
81 }
82 }
83 return Ok();
84 }
