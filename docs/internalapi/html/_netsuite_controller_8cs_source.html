157 [Route("v1/netsuite/salesorder")]
158 [HttpPost]
159 public IActionResult PostSalesOrder([FromBody] IncomingSalesOrderRequest soRequest)
160 {
161 try
162 {
163 _logger.LogInformation(LogService.LogSources.NetsuiteController, $"Received Netsuite POST for salesorder endpoint.", soRequest);
164 //get or create customer
165 NetsuiteCustomerRequest netsuiteCustomerRequest = _netsuiteHelper.BuildNetsuiteCustomerRequest(soRequest);
166 CustomerItemResponse customer = _netsuiteHelper.FindCustomerID(netsuiteCustomerRequest);
167 if (customer == null || customer.id == 0)
168 {
169 _netsuiteHelper.SendSalesOrderEmail(soRequest, 0, 0, 0);
170 return BadRequest("Customer not found or created");
171 }
172 //build sales order
173 NetsuiteSalesOrderRequest netsuiteSoRequest = _netsuiteHelper.BuildNetsuiteSalesOrderRequest(soRequest);
174 netsuiteSoRequest.entity = new IDRef() { id = customer.id };
175 netsuiteSoRequest.custbody_ab_internal_comments += $"\n{customer.custentity_ab_internal_comment}";
176 //post sales order
177 var salesOrderId = _netsuiteHelper.PostSalesOrderRequest(netsuiteSoRequest);
178 if (salesOrderId == 0)
179 {
180 _netsuiteHelper.SendSalesOrderEmail(soRequest, customer.id, 0, 0);
181 return BadRequest($"Sales order not created for customer {customer.id}");
182 }
183 //add payment if payment made
184 var depositId = 0;
185 if (soRequest.OrderDetails.DepositTotal > 0)
186 {
187 depositId = _netsuiteHelper.PostDeposit(salesOrderId, soRequest.OrderDetails.DepositTotal);
188 if (depositId == 0)
189 {
190 _netsuiteHelper.SendSalesOrderEmail(soRequest, customer.id, salesOrderId, 0);
191 return BadRequest($"Deposit not created for sales order {salesOrderId}");
192 }
193 }
194 //send final admin email
195 _netsuiteHelper.SendSalesOrderEmail(soRequest, customer.id, salesOrderId, depositId);
196 return Ok(salesOrderId);
197 }
198 catch (Exception ex)
199 {
200 _logger.LogException(LogService.LogSources.NetsuiteHelper,
201 $"Exception creating Netsuite sales order from controller",
202 ex.ToString());
203 return BadRequest($"Exception: {ex.ToString()}");
204 }

210 [Route("v1/netsuite/order")]
211 [HttpPost]
212 public IActionResult PostOrder([FromBody] IncomingNetsuiteOrderRequest oRequest)
213 {
214 try
215 {
216 _logger.LogInformation(LogService.LogSources.NetsuiteController, $"Received Netsuite POST for new salesorder endpoint.", oRequest);
217 if (oRequest.NetsuiteCustomer == null || (oRequest.NetsuiteSalesOrder == null && oRequest.NetsuitePurchaseOrder == null))
218 return BadRequest("Data submitted is bad");
219 //get or create customer
220 CustomerItemResponse customer = _netsuiteHelper.FindCustomerID(oRequest.NetsuiteCustomer);
221 if (customer == null || customer.id == 0)
222 return BadRequest("Customer not found or created");
223 StringBuilder orderError = new StringBuilder();
224 //if sales order exist
225 int salesOrderId = 0;
226 if (oRequest.NetsuiteSalesOrder != null)
227 {
228 //build sales order
229 oRequest.NetsuiteSalesOrder.entity = new IDRef() { id = customer.id };
230 oRequest.NetsuiteSalesOrder.custbody_ab_internal_comments += $"\n{customer.custentity_ab_internal_comment}";
231 //post sales order
232 salesOrderId = _netsuiteHelper.PostSalesOrderRequest(oRequest.NetsuiteSalesOrder);
233 if (salesOrderId == 0)
234 orderError.AppendLine($"Sales order not created for customer {customer.id}");
235 }
236 //if purchase order exist
237 int purchaseOrderId = 0;
238 if (oRequest.NetsuitePurchaseOrder != null)
239 {
240 //build purchase order
241 oRequest.NetsuitePurchaseOrder.entity = new IDRef() { id = customer.id };
242 oRequest.NetsuitePurchaseOrder.custbody_ab_internal_comments += $"\n{customer.custentity_ab_internal_comment}";
243 //post purchase order
244 purchaseOrderId = _netsuiteHelper.PostPurchaseOrderRequest(oRequest.NetsuitePurchaseOrder);
245 if (purchaseOrderId == 0)
246 orderError.AppendLine($"Sales order not created for customer {customer.id}");
247 }
248 if (orderError.Length > 0)
249 return BadRequest(orderError.ToString());
250 return Ok(new { CustomerID = customer.id, SalesOrderID = salesOrderId, PurchaseOrderID = purchaseOrderId });
251 }
252 catch (Exception ex)
253 {
254 _logger.LogException(LogService.LogSources.NetsuiteHelper,
255 $"Exception creating Netsuite sales order from controller",
256 ex.ToString());
257 return BadRequest($"Exception: {ex.ToString()}");
258 }
