47 [Route("v1/coinspot/{account}/{**endpoint}")]

49 public ActionResult<object> PassThrough([FromRoute] string account, [FromRoute] string endpoint, [FromBody] object body)
50 {
51 _logger.LogInformation($"Received Coinspot POST pass through for account {account}, endpoint '{endpoint}'.");
52
53 // Find Coinspot account settings
54 var coinspotAccount = _coinspotSettings?.Accounts?.FirstOrDefault(a => String.Equals(a.AccountName, account, StringComparison.CurrentCultureIgnoreCase));
55
56 if (coinspotAccount == null || string.IsNullOrEmpty(coinspotAccount.APIKey) || string.IsNullOrEmpty(coinspotAccount.APISecret))
57 {
58 var errorMessage = $"Coinspot endpoint not configured for account '{account}'";
59 _logger.LogError(errorMessage);
60 return NotFound($"Internal API: {errorMessage}");
61 }
62
63 JObject joBody = JObject.Parse(@"{'nonce': '" + DateTimeOffset.UtcNow.ToUnixTimeMilliseconds().ToString() + "'}");
64 joBody.Merge(body);
65
66 var client = new RestClient(_coinspotSettings.BaseURL);
67 RestRequest request = new RestRequest(endpoint, Method.POST);
68 request.AddJsonBody(JsonConvert.SerializeObject(joBody));
69 request.AddHeader("key", coinspotAccount.APIKey);
70 request.AddHeader("sign", SignData(coinspotAccount, Encoding.UTF8.GetBytes(JsonConvert.SerializeObject(joBody))).ToLower());
71 request.AddHeader("Content-Type", "application/json");
72
73 string responseContent = string.Empty;
74 IRestResponse response;
75 try
76 {
77 response = client.Execute(request);
78 if (response.StatusCode == HttpStatusCode.OK)
79 {
80 responseContent = response.Content;
81 }
82 else
83 {
84 return StatusCode(((int)response.StatusCode), $"{response.Content}");
85 }
86 }
87 catch (Exception ex)
88 {
89 return StatusCode(500, $"{ex}");
90 }
91 return Ok(responseContent);
92 }

102 private string SignData(CoinspotAccount coinspotAccount, byte[] JSONData)
103 {
104 var HMAC = new HMACSHA512(Encoding.UTF8.GetBytes(coinspotAccount.APISecret));
105 byte[] EncodedBytes = HMAC.ComputeHash(JSONData);
106
107 StringBuilder stringBuilder = new StringBuilder();
108
109 for (int i = 0; i <= EncodedBytes.Length - 1; i++)
110 stringBuilder.Append(EncodedBytes[i].ToString("X2"));
111
112 return stringBuilder.ToString();
113 }
