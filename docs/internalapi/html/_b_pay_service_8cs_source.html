138 private bool ValidateCustomerID(BPayCustomer Customer)
139 {
140 if (Customer.CustomerSource == CustomerSource.NetsuiteBullionBNE.ToString())
141 {
142 if (Customer.CustomerID.Length <= 6)
143 {
144 var LeadingNums = "";
145 for (var i = 1; i <= 6 - Customer.CustomerID.Length; i += 1)
146 LeadingNums += (int)(CustomerSource)Enum.Parse(typeof(CustomerSource), Customer.CustomerSource);
147 Customer.CustomerID = $"{(int)(CustomerSource)Enum.Parse(typeof(CustomerSource), Customer.CustomerSource)}{LeadingNums}{Customer.CustomerID}";
148 }
149 else
150 {
151 _logger.LogError(LogService.LogSources.bPayService,
152 $"Customer ID over 6 digits in length, BPay not being generated",
153 "Customer object:",
154 Customer);
155 SendCustomerIDLengthTooLongEmail(Customer);
156 return false;
157 }
158 }
159 else
160 {
161 return false;
162 }
163 return true;
