51 public PostCustomer FindCustomer(PostCustomer customer)

53 logger.LogDebug(LogService.LogSources.GreenIDDBHelper, $"searching for customer in local customers db", customer);
54 try
55 {
56 VerifiedClient customerMatch = null;
57 //fix apostrophes

61 {
62 //fisrt search on greenidref if supplied

64 .Where(c => c.GreenIdref.Trim() == customer.greenIdRef.Trim())
65 .OrderByDescending(c => c.DateAdded)
66 .FirstOrDefault();
67 //we have greenidref but if we cant find in database
68 if (customerMatch == null)
69 {
70 //look up green id for ab online
71 GreenIDCustomer greenIdCustomer = greenIDService.GetGreenIdCustomer("ab", customer.greenIdRef);
72 if (greenIdCustomer != null && greenIdCustomer.verificationResult.ToLower().Contains("verified"))

74 AddCustomer("ab", greenIdCustomer);

78 //we have greenidref but if we cant find in database or in ab online

84 {

88 }

90 if (customerMatch == null &&
91 !string.IsNullOrEmpty(customer.firstName) &&
92 !string.IsNullOrEmpty(customer.lastName) &&
93 !string.IsNullOrEmpty(customer.email))
94 {
95 //next search on email as most reliable
96 customerMatch = context.VerifiedClients
97 .Where(c => c.GivenName.Trim().ToLower() == customer.firstName.Trim().ToLower() &&
98 c.Surname.Trim().ToLower() == customer.lastName.Trim().ToLower() &&
99 c.Email.Trim().ToLower() == customer.email.Trim().ToLower())
100 .OrderByDescending(c => c.DateAdded)
101 .FirstOrDefault();
102 }
103 if (customerMatch == null &&
104 !string.IsNullOrEmpty(customer.firstName) &&
105 !string.IsNullOrEmpty(customer.lastName) &&

107 {
108 //next search on names matching, and suburb
109 var customerMatches = context.VerifiedClients
110 .Where(c => c.GivenName.Trim().ToLower() == customer.firstName.Trim().ToLower()
111 && c.Surname.Trim().ToLower() == customer.lastName.Trim().ToLower()
112 && c.StreetName.Trim().ToLower() == customer.suburb.Trim().ToLower())
113 .OrderByDescending(c => c.DateAdded);
114 //matched customers, so update with email address from query if exists
115 if (customerMatches != null && customerMatches.Count() > 0 && !string.IsNullOrEmpty(customer.email))
116 {
117 foreach (VerifiedClient vc in customerMatches)

119 vc.Email = customer.email;
120 context.Entry(vc).State = Microsoft.EntityFrameworkCore.EntityState.Modified;
121 }
122 context.SaveChanges();
123 customerMatch = customerMatches.First();
124 }
125 }

184 private string FixApostrophe(string inputString)
185 {
186 char[] badApostrophes = "’‘`".ToCharArray();
187 if (!string.IsNullOrEmpty(inputString) && inputString.IndexOfAny(badApostrophes) > -1)
188 inputString = inputString.Replace("’", "'").Replace("‘", "'").Replace("`", "'");
189 return inputString;
190 }
