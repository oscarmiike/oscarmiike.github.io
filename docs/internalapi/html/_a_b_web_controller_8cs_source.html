40 [HttpGet("trimlogs")]
41 public ActionResult<object> TrimLogs([FromQuery] int keepDays = 0)
42 {

44 keepDays = 30;
45 var trimDate = DateTime.Now.AddDays(-keepDays);
46 var deleteSql = $"delete from ablogentry where created < '{trimDate.Year}-{trimDate.Month}-{trimDate.Day}'";
47 var deletedLogs = 0;

50 {
51 deletedLogs = _context.Database.ExecuteSqlRaw(deleteSql);

53 catch (Exception ex)
