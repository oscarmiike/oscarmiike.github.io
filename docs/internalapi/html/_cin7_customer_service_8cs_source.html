44 public GetCin7CustomerResponse GetOrCreateCustomer(CreateCin7CustomerRequest createCustomerRequest, CustomerFoundBehaviour customerFoundBehaviour = CustomerFoundBehaviour.DoNotUpdate)
45 {
46 var surnameSearchResult = SearchName(createCustomerRequest.AinslieCompanyOption, createCustomerRequest.Surname, createCustomerRequest.FirstName)?.Where(c => CustomerMatchesSearchResult(c, createCustomerRequest)).ToList();
47
48 _logger.LogDebug($"Found {surnameSearchResult.Count} possible customers from {createCustomerRequest.AinslieCompanyOption} - {string.Join(", ", surnameSearchResult.Select(r => $"Id {r.Cin7Id} : {r.FirstName} {r.Surname}"))}");
49
50 if (surnameSearchResult?.Any() == true)
51 {
52 // Prioritise green id (1 point) and bpay reference details (2 points):
53 // Record with GreenId only: 1 point
54 // Record with BPay only: 2 points
55 // Record with both fields: 3 points
56 // Record with none: 0 points
57 // Order descending
58 // Take first
59 var matchingRecord = surnameSearchResult.OrderByDescending(c =>
60 (string.IsNullOrEmpty(c.GreenIdReference) ? 0 : 1) + (string.IsNullOrEmpty(c.BPayReference) ? 0 : 2))
61 .First();
62
63 _logger.LogDebug($"Matched to customer from {createCustomerRequest.AinslieCompanyOption}: {JsonConvert.SerializeObject(matchingRecord)}");
64
65 if (customerFoundBehaviour == CustomerFoundBehaviour.Update)
66 {
67 // Perform update in Cin7:
68 throw new NotImplementedException($"Update currently not implemented.");
69 }
70
71 // Return existing customer
72 return _mapper.Map<GetCin7CustomerResponse>(matchingRecord);
73 // return Cin7CustomerToCustomerResponse(matchingRecord);
74 }
75
76 // Create new customer
77 return CreateCin7Customer(createCustomerRequest);
78 }

131 // compare by upper, trimming, and removing white space:
132 // surname, firstname and one of either email or phone (phone or mobile)
133 private bool CustomerMatchesSearchResult(GetCin7CustomerResponse searchCommand, CreateCin7CustomerRequest createSalesOrderCommand)
134 {
135 if (string.IsNullOrWhiteSpace(searchCommand.FirstName) || searchCommand.FirstName?.ToUpper()?.Replace(" ", "") != createSalesOrderCommand.FirstName.ToUpper().Replace(" ", ""))
136 return false;
137
138 if (string.IsNullOrWhiteSpace(searchCommand.Surname) || searchCommand.Surname.ToUpper().Replace(" ", "") != createSalesOrderCommand.Surname.ToUpper().Replace(" ", ""))
139 return false;
140
141 if (!string.IsNullOrWhiteSpace(searchCommand.Email) && searchCommand.Email?.ToUpper()?.Replace(" ", "") == createSalesOrderCommand.Email?.ToUpper()?.Replace(" ", ""))
142 return true;
143
144 if (!string.IsNullOrWhiteSpace(searchCommand.Mobile) && searchCommand.Mobile?.ToUpper()?.Replace(" ", "") == createSalesOrderCommand.Phone?.ToUpper()?.Replace(" ", ""))
145 return true;
146
147 if (!string.IsNullOrWhiteSpace(searchCommand.Phone) && searchCommand.Phone?.ToUpper()?.Replace(" ", "") == createSalesOrderCommand.Phone?.ToUpper()?.Replace(" ", ""))
148 return true;
149
150 return false;
151 }

153 public GetCin7CustomerResponse CreateCin7Customer(CreateCin7CustomerRequest createCustomerCommand)
154 {
155 var client = GetOrCreateRestClient(createCustomerCommand.AinslieCompanyOption);
156
157 // Create new Cin7 contact
158
159 var emailResults = SearchEmail(createCustomerCommand.AinslieCompanyOption, createCustomerCommand.Email);
160
161 var addEmail = emailResults != null && !emailResults.Any();
162
163 CreateContact contactToCreate = new CreateContact()
164 {
165 firstName = createCustomerCommand.FirstName.Trim(),
166 lastName = createCustomerCommand.Surname.Trim(),
167 type = "customer",
168 company = !string.IsNullOrEmpty(createCustomerCommand.Company.Trim()) ? createCustomerCommand.Company.Trim() : createCustomerCommand.FirstName.Trim() + " " + createCustomerCommand.Surname.Trim(),
169 email = addEmail ? createCustomerCommand.Email.Trim() : null,
170 phone = createCustomerCommand.Phone.Trim(),
171 address1 = createCustomerCommand.StreetAddress.Trim(),
172 city = createCustomerCommand.Suburb.Trim(),
173 state = createCustomerCommand.State.Trim(),
174 postCode = createCustomerCommand.PostCode.Trim(),
175 postalAddress1 = createCustomerCommand.StreetAddress.Trim(),
176 postalCity = createCustomerCommand.Suburb.Trim(),
177 postalState = createCustomerCommand.State.Trim(),
178 postalPostCode = createCustomerCommand.PostCode.Trim()
179 };
180
181 _logger.LogDebug($"Creating a new customer in {createCustomerCommand.AinslieCompanyOption}: {JsonConvert.SerializeObject(contactToCreate)}");
182
183 List<CreateContact> contactList = new List<CreateContact> { contactToCreate };
184
185 var request = new RestRequest("/v1/contacts", Method.POST);
186 var requestJson = JsonConvert.SerializeObject(contactList);
187 request.AddParameter("application/json", requestJson, ParameterType.RequestBody);
188
189 var responseMessage = client.ExecuteWithRetry(request, 3, 5000)
190 .Parse<IEnumerable<Cin7ResponseItem>>();
191
192 // To remain under 1 request per second Cin7 limit:
193 Task.Delay(5000).GetAwaiter().GetResult();
194
195 if (!responseMessage.IsSuccessful || responseMessage.Data == null)
196 throw new InvalidOperationException($"Error creating a contact in Cin7: Status {responseMessage.StatusDescription} : {responseMessage.Content}");
197
198 var responseList = responseMessage.Data;
199
200 if (responseList == null || responseList.Count() == 0)
201 throw new InvalidOperationException($"No contact responses returned from Cin7. Creation failed.");
202
203 var response = responseList.FirstOrDefault();
204
205 if (response.success != true)
206 throw new InvalidOperationException($"Failed to create contact in Cin7: {string.Join(" ", response.errors)}");
207
208 // As we've just created the customer, return these values. We can trust that these values are now correct
209 // (we are saving a call to Cin7)
210 return new GetCin7CustomerResponse()
211 {
212 Cin7Id = response.id,
213 Surname = createCustomerCommand.Surname.Trim(),
214 FirstName = createCustomerCommand.FirstName.Trim(),
215 Phone = createCustomerCommand.Phone.Trim(),
216 Email = createCustomerCommand.Email.Trim(),
217 Company = createCustomerCommand.Company,
218 Address1 = createCustomerCommand.StreetAddress,
219 Suburb = createCustomerCommand.Suburb,
220 PostCode = createCustomerCommand.PostCode,
221 State = createCustomerCommand.State,
222 PostalAddress1 = createCustomerCommand.StreetAddress,
223 PostalSuburb = createCustomerCommand.Suburb,
224 PostalPostCode = createCustomerCommand.PostCode,
225 PostalState = createCustomerCommand.State
226 };
227 }
