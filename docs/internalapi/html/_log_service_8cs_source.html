94 public LogEntry InsertLog(LogTypes logType, int threadID, LogSources logSource, params object[] DataList)
95 {
96 LogEntry logEntry;
97 try
98 {
99 //try to create log entry by deserialising objects in param array datalist
100 logEntry = new LogEntry()
101 {
102 Created = DateTime.Now,
103 Type = logType.ToString(),
104 ThreadID = threadID,
105 Source = logSource.ToString(),
106 Data = JsonConvert.SerializeObject(DataList, new JsonSerializerSettings
107 {
108 ReferenceLoopHandling = ReferenceLoopHandling.Ignore
109 })
110 };
111 }
112 catch (Exception ex)
113 {
114 //create new logentry with exception
115 logEntry = new LogEntry()
116 {
117 Created = DateTime.Now,
118 Type = LogTypes.Exception.ToString(),
119 ThreadID = threadID,
120 Source = LogSources.LogService.ToString(),
121 Data = $"Exception creating logEntry: {ex.ToString()}"
122 };
123 }
124 try
125 {
126 _context.LogEntries.Add(logEntry);
127 _context.SaveChanges();
128 return logEntry;
129 }
130 catch (Exception ex)
131 {
132 WriteLog(LogTypes.Exception, "LogService", $"Exception trying to insert log. Exception: {ex.ToString()}");
133 }
134 return null;
135 }

140 private void WriteLog(LogTypes Type, string Source, string Description)
141 {
142 var LogPath = $"{AppDomain.CurrentDomain.BaseDirectory}/Log/";
143 if (!Directory.Exists(LogPath))
144 Directory.CreateDirectory(LogPath);
145 var LogFile = $"InternalAPI.Log";
146 try
147 {
148 StreamWriter SW = File.AppendText(LogPath + LogFile);
149 SW.WriteLine($"{DateTime.Now.ToString()} [{Type.ToString()}] {Description}");
150 SW.Close();

153 {
154 }
