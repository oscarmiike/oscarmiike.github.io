50 public ActionResult<WebShopResponse> Sales(string branch, string start, string finish)
51 {
52 _logger.LogDebug(LogService.LogSources.WebShopController, $"WebShop/Sales request received. TimeBoundsUTC: {start} - {finish}");
53 List<WebShopSale> WebShopSales = new List<WebShopSale>();
54 DateTime startTime;
55 DateTime finishTime;
56 if (string.IsNullOrEmpty(start) || string.IsNullOrEmpty(finish) || !DateTime.TryParse(start, out startTime) || !DateTime.TryParse(finish, out finishTime))
57 {
58 _logger.LogError(LogService.LogSources.WebShopController, $"Webshop sales request startdate or finishdate not supplied correctly. Start:{start} Finish:{finish}");
59 webShopResponse.success = false;
60 webShopResponse.errors.Add($"Webshop sales request startdate or finishdate not supplied correctly. Start:{start} Finish:{finish}");
61 return webShopResponse;
62 }
63 if (startTime > finishTime)
64 {
65 _logger.LogError(LogService.LogSources.WebShopController, $"Webshop sales request startdate is greater than finishdate. Start:{start} Finish:{finish}");
66 webShopResponse.success = false;
67 webShopResponse.errors.Add($"Webshop sales request startdate is greater than finishdate. Start:{start} Finish:{finish}");
68 return webShopResponse;
69 }
70
71 var startTimeLocal = startTime.ToLocalTime();
72 var finishTimeLocal = finishTime.ToLocalTime();
73
74 //get orders
75 //take orders before the starttime to see if there were any duplicates prior to start time -we remove this extra time later
76 var orderHeaders = _context.OrderHeaders.Where(oh => oh.BranchOffice.ToLower() == branch.ToLower() && oh.OrderTime >= startTimeLocal.AddMinutes(-duplicateThresholdMin) && oh.OrderTime < finishTimeLocal && oh.OrderPaymentResult != null).OrderBy(oh => oh.OrderTime).ToList();
77 if (orderHeaders == null || orderHeaders.Count() < 1)
78 {
79 webShopResponse.success = true;
80 return webShopResponse;
81 }
82 var orderItems = _context.OrderItems.Where(oi => orderHeaders.Select(oh => oh.OrderId).Contains(oi.OrderId)).ToList();
83 if (orderItems == null || orderItems.Count() < 1)
84 {
85 _logger.LogError(LogService.LogSources.WebShopController, $"Webshop sales request returned orders but no items");
86 webShopResponse.success = false;
87 webShopResponse.errors.Add($"Webshop sales request returned orders but no items");
88 return webShopResponse;
89 }
90
91 //check duplicate customers in this timeframe, get list of emails that have duplicate orders for validation
92 var duplicateCustomerEmails = orderHeaders.GroupBy(oh => oh.CustomerEmail.ToLower()).Where(grp => grp.Count() > 1).Select(group => new { email = group.Key, count = group.Count() });
93 if(duplicateCustomerEmails.Count() > 0)
94 {
95 //group orders for duplicate customers
96 var duplicateCustomerOrders = orderHeaders.GroupBy(oh => oh.CustomerEmail.ToLower()).Where(grp => grp.Count() > 1);
97 foreach (var grp in duplicateCustomerOrders)
98 {
99 bool misMatchedItemFound = false;
100 //group order items for this group of duplicate customer orders. one group of order items for each order ID
101 var duplicateCustomerOrderItems = orderItems.Where(oi => grp.Any(grpoh => grpoh.OrderId == oi.OrderId)).GroupBy(oi => oi.OrderId);
102 //compare each iteration to first()
103 foreach (var oiGrp in duplicateCustomerOrderItems)
104 {
105 if (misMatchedItemFound) break;
106 var firstOrderItemSet = duplicateCustomerOrderItems.First();
107 foreach (var oi in oiGrp)
108 {
109 if (!firstOrderItemSet.Any(oiGrpItem => oiGrpItem.ItemSku == oi.ItemSku && oiGrpItem.ItemQty == oi.ItemQty))
110 {
111 misMatchedItemFound = true;
112 }
113 }
114 }
115 //if itemskus and itemqtys from all orders matched each other
116 if (!misMatchedItemFound)
117 {
118 //check if the first order was outside of the current time scope
119 if (grp.OrderBy(oh => oh.OrderTime).First().OrderTime < startTimeLocal)
120 {
121 //remove all orders with this email address because we only take first order in a group of duplicates and it was outside this time scope
122 orderHeaders.RemoveAll(oh => oh.CustomerEmail.ToLower() == grp.First().CustomerEmail.ToLower());
123 }
124 else
125 {
126 //remove everything other than the first order because we take first order in a group of duplicates
127 orderHeaders.RemoveAll(oh => oh.CustomerEmail.ToLower() == grp.First().CustomerEmail.ToLower() && oh.OrderTime > grp.OrderBy(ohGrp => ohGrp.OrderTime).First().OrderTime);
128 }
129 }
130 else
131 {
132 //mismatched items found, assume all orders are ok
133 }
134 }
135 }
136
137 if (productList == null || productList.Count < 1)
138 {
139 webShopResponse.success = false;
140 webShopResponse.errors.Add("Priceapp product list unavailable");
141 return webShopResponse;
142 }
143 //strip orderheaders duplicate order threshold orders, trim down to request time bounds
144 foreach (OrderHeader oh in orderHeaders.Where(oh => oh.OrderTime >= startTimeLocal))
145 {
146 // Create the order header DTO (data transfer object)
147 // This differs from the database model, as the database datetimes are local time
148 // and the DTO is in UTC
149 var ohDto = (OrderHeader)oh.Clone();
150 ohDto.OrderTime = oh.OrderTime.ToUniversalTime();
151
152 var orderHeaderItems = orderItems.Where(oi => oi.OrderId == oh.OrderId).ToList();
153 WebShopSales.Add(new WebShopSale()
154 {
155 orderHeader = ohDto,
156 orderItems = orderHeaderItems,
157 orderWeights = CalculateOrderWeight(orderHeaderItems)
158 }); ;
159 }
160 webShopResponse.responseObject = WebShopSales;
161
162 if (webShopResponse.errors.Count > 0)
163 {
164 webShopResponse.success = false;
165 }
166 else
167 {
168 webShopResponse.success = true;
169 }

178 List<OrderWeight> CalculateOrderWeight(List<OrderItem> orderItems)
179 {
180 List<OrderWeight> orderWeights = new List<OrderWeight>();
181 foreach(OrderItem oi in orderItems)
182 {
183 Product pr = productList.Where(pr => pr.SKU == oi.ItemSku && pr.HasWeight && !pr.BaseAsset.IsHardwareAsset).FirstOrDefault();
184 if (pr == null)
185 {
186 webShopResponse.warnings.Add($"SKU {oi.ItemSku} not found in PriceApp");
187 continue;
188 }
189 var orderWeightAsset = orderWeights.Where(ow => ow.AssetCode == pr.BaseAsset.AssetCode).FirstOrDefault();
190 if(orderWeightAsset == null)
191 {
192 orderWeightAsset = new OrderWeight()
193 {
194 AssetCode = pr.BaseAsset.AssetCode,
195 Weight=0
196 };
197 orderWeights.Add(orderWeightAsset);
198 }
199 var productWeightOz = ConvertWeight(pr.NetWeight, pr.NetWeightUnit, orderWeightAsset.WeightUnit);
200 orderWeightAsset.Weight += oi.ItemQty * productWeightOz;
201 }
202 return orderWeights;
203 }

205 public static double ConvertWeight(double Val, Spotunit InputUnit, Spotunit ReturnUnit)
206 {
207 if (InputUnit == null | ReturnUnit == null || (!InputUnit.HasWeight | !ReturnUnit.HasWeight))
208 return Val;
209 Val = Val * (InputUnit.Grams / (double)ReturnUnit.Grams);
210 return Val;
