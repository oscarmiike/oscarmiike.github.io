107 public static IRestResponse ExecuteWithRetry(this IRestClient client, IRestRequest request, int maxRetries = 3, int initialBackOffMilliseconds = 1000)
108 {
109 IRestResponse response = default;
110 for (int i = 0; i <= Math.Max(0, maxRetries); i++)
111 {
112 response = client.Execute(request);
113 if ((int)response.StatusCode != 429)
114 return response;
115
116 // 5 second delay, exponentially increasing
117 Task.Delay(initialBackOffMilliseconds * Convert.ToInt32(Math.Pow(2, i))).GetAwaiter().GetResult();
118 }
119
120 return response;
121 }
