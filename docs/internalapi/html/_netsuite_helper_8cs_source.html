102 public NetsuiteSalesOrderRequest BuildNetsuiteSalesOrderRequest(IncomingSalesOrderRequest incomingRequest)
103 {
104 NetsuiteSalesOrderRequest soRequest = new NetsuiteSalesOrderRequest();
105 soRequest.trandate = DateTime.Now.ToString("yyyy-MM-dd");
106 if (soRequest.customform == 0)
107 soRequest.customform = _netsuiteSettings.SalesOrderForm;
108 soRequest.subsidiary = (int)Enum.Parse<NetsuiteSubsidiaries>(incomingRequest.CustomerSubsidiary);
109 soRequest.custbody_ab_project = new NameRef() { refName = incomingRequest.NetsuiteProject };
110 soRequest.custbody_ab_stage_so = new NameRef() { refName = incomingRequest.Stage };
111 soRequest.custbody_ab_channel = new NameRef() { refName = "Web" };
112 if (incomingRequest.Stage.ToLower().Contains("ste"))
113 {
114 //shipping
115 soRequest.location = new NameRef() { refName = $"STE - {incomingRequest.CustomerSubsidiary}" };
116 soRequest.shipmethod = new NameRef() { refName = $"Courier - STE - {incomingRequest.CustomerSubsidiary}" };
117 soRequest.shippingcost = incomingRequest.OrderDetails.ShippingTotal;
118 soRequest.shipaddress =
119 $"{incomingRequest.CustomerDetails.FirstName} {incomingRequest.CustomerDetails.Surname}\n" +
120 $"{incomingRequest.CustomerDetails.StreetAddress}\n" +
121 $"{incomingRequest.CustomerDetails.Suburb} {incomingRequest.CustomerDetails.State} {incomingRequest.CustomerDetails.PostCode}\n" +
122 $"Australia";
123 }
124 else
125 {
126 //not shipping (hold, deliver to vault, storage)
127 soRequest.location = new NameRef() { refName = $"{NetsuiteSubsidiaryToLocationName(Enum.Parse<NetsuiteSubsidiaries>(incomingRequest.CustomerSubsidiary))}"};
128 if (incomingRequest.Stage.ToLower().Contains("hold"))
129 {
130 soRequest.shipmethod = new NameRef() { refName = $"In Store - {incomingRequest.CustomerSubsidiary}" };
131 }
132 if (incomingRequest.Stage.ToLower().Contains("deliver to"))
133 {
134 soRequest.shipmethod = new NameRef() { refName = $"Deliver to Vault - {incomingRequest.CustomerSubsidiary}" };
135 }
136 if (incomingRequest.Stage.ToLower().Contains("storage"))
137 {
138 soRequest.shipmethod = new NameRef() { refName = $"Storage - {incomingRequest.CustomerSubsidiary}" };
139 }
140 }
141 soRequest.otherRefNum = incomingRequest.SalesOrderReference;
142 soRequest.custbodydelivery_instructions += $"{incomingRequest.Stage}{Environment.NewLine}{incomingRequest.DeliveryInstructions}";
143 soRequest.custbody_ab_internal_comments += $"{incomingRequest.InternalComments}";
144 soRequest.item = new NetsuiteSalesOrderRequest.ItemList()
145 {
146 items = new List<NetsuiteSalesOrderRequest.ItemData>()
147 };
148 foreach (IncomingSalesOrderLineItem x in incomingRequest.OrderDetails.OrderLines)
149 {
150 soRequest.item.items.Add(
151 new NetsuiteSalesOrderRequest.ItemData()
152 {
153 item = new NameRef()
154 {
155 refName = x.ProductSKU
156 },
157 quantity = x.Qty,
158 rate = x.QtyPrice,
159 description = $"{x.ProductName}{(string.IsNullOrEmpty(x.ProductWarning) ? "" : $" - {x.ProductWarning}")}",
160 location = new NameRef() { refName = $"{NetsuiteSubsidiaryToLocationName(Enum.Parse<NetsuiteSubsidiaries>(incomingRequest.CustomerSubsidiary))}" }
161 }
162 );
163 }
164 return soRequest;
165 }
