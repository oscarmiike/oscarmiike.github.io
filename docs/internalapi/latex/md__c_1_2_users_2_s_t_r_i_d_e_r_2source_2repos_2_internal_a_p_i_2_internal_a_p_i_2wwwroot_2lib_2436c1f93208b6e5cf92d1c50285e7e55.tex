\DoxyCodeLine{export\ default\ [}

\DoxyCodeLine{\ \ \ \ compat.extends("{}standard"{},\ "{}example"{}),}
\DoxyCodeLine{}
\DoxyCodeLine{\ \ \ \ //\ mimic\ environments}
\DoxyCodeLine{\ \ \ \ compat.env(\{}

\DoxyCodeLine{\ \ \ \ \ \ \ \ node:\ true}
